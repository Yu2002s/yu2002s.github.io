简单语法

class BrvActivity: AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        val binding = ActivityBrvBinding.inflate(layoutInflater)
        setContentView(binding.root)

        val rv = binding.rv
        rv.linear().setup {}
    }
}
